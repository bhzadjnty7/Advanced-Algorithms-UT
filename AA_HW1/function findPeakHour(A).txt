function findPeakHour(A):
    n = length(A)
    left = 0
    right = n - 1
    
    while left < right:
        mid = left + (right - left) / 2
        
        // Check if mid is a peak
        if (mid == 0 or A[mid] > A[mid-1]) and (mid == n-1 or A[mid] > A[mid+1]):
            return mid
        
        // If ascending, the peak is on the right side
        else if mid < n-1 and A[mid] < A[mid+1]:
            left = mid + 1
        
        // Otherwise, the peak is on the left side
        else:
            right = mid
    
    return left  // Finally left equals right
